18  package com.agorapulse.micronaut.bigquery.impl;
19  
20  import com.agorapulse.micronaut.bigquery.BigQueryService;
21  import com.agorapulse.micronaut.bigquery.RowResult;
22  import com.google.cloud.bigquery.BigQuery;
23  import com.google.cloud.bigquery.BigQueryException;
24  import com.google.cloud.bigquery.Job;
25  import com.google.cloud.bigquery.JobId;
26  import com.google.cloud.bigquery.JobInfo;
27  import com.google.cloud.bigquery.QueryJobConfiguration;
28  import com.google.cloud.bigquery.QueryParameterValue;
29  import com.google.cloud.bigquery.StandardSQLTypeName;
30  import com.google.cloud.bigquery.TableResult;
31  import io.reactivex.Flowable;
32  
33  import javax.inject.Singleton;
34  import java.time.Instant;
35  import java.util.Collections;
36  import java.util.LinkedHashMap;
37  import java.util.Map;
38  import java.util.UUID;
39  import java.util.function.Function;
40  
41  @Singleton
42  public class DefaultBigQueryService implements BigQueryService {
43  
44      private final BigQuery bigquery;
45  
46      public DefaultBigQueryService(BigQuery bigQuery) {
47          this.bigquery = bigQuery;
48      }
49  
50      @Override
51      public <T> Flowable<T> query(Map<String, ?> namedParameters, String sql, Function<RowResult, T> builder) {
52          QueryJobConfiguration queryConfig = QueryJobConfiguration
53              .newBuilder(checkForNulls(sql, namedParameters))
54              .setUseLegacySql(false)
55              .setNamedParameters(toNamedParameters(namedParameters))
56              .build();
57  
58          // Create a job ID
59          JobId jobId = JobId.of(UUID.randomUUID().toString());
60          Job job = bigquery.create(JobInfo.newBuilder(queryConfig).setJobId(jobId).build());
61  
62          try {
63              // Wait for the query to complete.
64              Job completedJob = job.waitFor();
65  
66              // Check for errors
67              if (completedJob == null) {
68                  throw new IllegalStateException("Job no longer exists");
69              } else if (completedJob.getStatus().getError() != null) {
70                  throw new IllegalStateException("Failed to execute sql " + sql + ":" + completedJob.getStatus().getError());
71              }
72  
73  
74              TableResult result = completedJob.getQueryResults();
75              return Flowable.fromIterable(result.iterateAll()).filter(r -> !r.isEmpty()).map(FieldValueListRowResult::new).map(builder::apply);
76          } catch (InterruptedException | BigQueryException e) {
77              throw new IllegalStateException("Could not execute query: " + sql, e);
78          }
79      }
80  
81      @Override
82      public void execute(Map<String, ?> namedParameters, String sql) {
83          QueryJobConfiguration queryConfig = QueryJobConfiguration
84              .newBuilder(checkForNulls(sql, namedParameters))
85              .setUseLegacySql(false)
86              .setNamedParameters(toNamedParameters(namedParameters))
87              .build();
88  
89          // Create a job ID
90          JobId jobId = JobId.of(UUID.randomUUID().toString());
91          Job job = bigquery.create(JobInfo.newBuilder(queryConfig).setJobId(jobId).build());
92  
93          try {
94              // Wait for the query to complete.
95              Job completedJob = job.waitFor();
96  
97              // Check for errors
98              if (completedJob == null) {
99                  throw new IllegalStateException("Job no longer exists");
100             } else if (completedJob.getStatus().getError() != null) {
101                 throw new IllegalStateException(completedJob.getStatus().getError().toString());
102             }
103         } catch (InterruptedException | BigQueryException e) {
104             throw new IllegalStateException("Could not execute sql: " + sql, e);
105         }
106     }
107 
108     @SuppressWarnings("unchecked")
109     private Map<String, QueryParameterValue> toNamedParameters(Map<String, ?> namedParameters) {
110         if (namedParameters.isEmpty()) {
111             return Collections.emptyMap();
112         }
113 
114         final Map<String, QueryParameterValue> result = new LinkedHashMap<>();
115 
116         namedParameters.forEach((key, value) -> {
117             Object converted = convertIfNecessary(value);
118             if (converted instanceof Instant) {
119                 result.put(key, QueryParameterValue.of(((Instant) converted).toEpochMilli(), StandardSQLTypeName.TIMESTAMP));
120             } else if (converted != null){
121                 result.put(key, QueryParameterValue.of(converted, (Class<Object>) converted.getClass()));
122             }
123         });
124 
125         return result;
126     }
127 
128     private String checkForNulls(String sql, Map<String, ?> namedParameters) {
129         String result = sql;
130         for(Map.Entry<String, ?> entry : namedParameters.entrySet()) {
131             if (entry.getValue() == null) {
132                 result = result.replace("@" + entry.getKey(), "null");
133             }
134         }
135         return result;
136     }
