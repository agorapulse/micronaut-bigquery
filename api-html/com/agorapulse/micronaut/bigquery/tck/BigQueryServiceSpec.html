018 package com.agorapulse.micronaut.bigquery.tck
019 
020 import com.agorapulse.micronaut.bigquery.BigQueryService
021 import io.micronaut.context.ApplicationContext
022 import spock.lang.AutoCleanup
023 import spock.lang.Specification
024 import spock.lang.Unroll
025 
026 @SuppressWarnings([
027     'AbstractClassWithoutAbstractMethod',
028     'DuplicateStringLiteral',
029     'DuplicateNumberLiteral',
030 ])
031 abstract class BigQueryServiceSpec extends Specification {
032 
033     // tag::test-setup[]
034     @AutoCleanup ApplicationContext context
035     // end::test-setup[]
036 
037     BigQueryService bigquery
038 
039     // tag::setup[]
040     void setup() {
041         context = buildContext()
042         context.start()
043 
044         bigquery = context.getBean(BigQueryService)
045     }
046     // end::setup[]
047 
048     abstract ApplicationContext buildContext()
049 
050     @Unroll
051     @SuppressWarnings('AbcMetric')
052     // tag::feature[]
053     void 'handle bigquery operations using #serviceType.simpleName'() {
054         given:
055             PersonService service = context.getBean(serviceType)
056             // end::feature[]
057         when:
058             service.deleteEverything()
059         then:
060             noExceptionThrown()
061 
062         when:
063             Person vlad = service.createPerson('Vladimir', 'Orany', 'vlad@agorapulse.com', Role.ADMIN)
064         then:
065             noExceptionThrown()
066 
067         when:
068             Optional<Person> loaded = service.get(vlad.id)
069         then:
070             loaded.present
071             vlad == loaded.get()
072             vlad.hashCode() == loaded.get().hashCode()
073             !vlad.is(loaded.get())
074 
075         when:
076             Optional<Person> unsafe = service.getUnsafe(vlad.id)
077         then:
078             unsafe.present
079             vlad == unsafe.get()
080             !vlad.is(unsafe.get())
081 
082         when:
083             Person luke = service.createPerson('Luke', 'Orany', 'luke@awesomeguy.com', Role.GUEST)
084         and:
085             List<Person> oranys = service.findByLastName('Orany').toList().blockingGet()
086             List<Person> oranysUnsafe = service.findByLastNameUnsafe('Orany').toList().blockingGet()
087         then:
088             oranys.size() == 2
089             luke in oranys
090 
091             oranysUnsafe.size() == 2
092             luke in oranysUnsafe
093 
094         when:
095             service.updateRole(vlad.id, Role.ADMIN)
096         then:
097             service.get(vlad.id).get().role == Role.ADMIN
098 
099         when:
100             service.deletePerson(vlad.id)
101         then:
102             !service.get(vlad.id).present
103             service.get(luke.id).present
104 
105         when:
106             service.findByLastNameUnsafe('\' bobby_tables').blockingFirst()
107         then:
108             thrown(RuntimeException)
109 
110         when:
111             Person bobby = service.createPerson('Booby', 'Tables', 'bobby@deletefromtables.com', Role.GUEST)
112             bigquery.execute('update persons.persons set role = \'noone\'')
113             service.get(bobby.id)
114         then:
115             thrown(RuntimeException)
116 
117         when:
118             bigquery.execute('some fake sql with :person', person: bobby)
119         then:
120             thrown(RuntimeException)
121 
122         when:
123             bigquery.execute('some fake sql')
124         then:
125             thrown(RuntimeException)
126 
127         when:
128             service.deleteEverything()
129         then:
130             !service.get(vlad.id).present
131             !service.get(luke.id).present
132         where:
133             serviceType << [
134                 JavaPersonService,
135                 GroovyPersonService,
136             ]
137     }
138 
139 }
